문제 5 ]

		empno 사원번호, RPAD(SUBSTR(ename, 1, 2), length(ename), '*')  사원이름, job 직급, TRUNC(MONTHS_BETWEEN(SYSDATE, hiredate)) 근무개월수
	FROM emp;
